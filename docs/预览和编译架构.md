# HoneyGUI Design 预览和编译架构

## 概述

本文档描述 HoneyGUI Design 的预览和编译架构，实现从可视化设计到 PC 模拟器预览的完整流程。

## 技术路线

```
设计器 → HML → C代码生成 → SCons编译 → 模拟器运行 → PC预览
```

**核心原理**：
- 设计器生成 HML 文件描述界面
- C 代码生成器将 HML 转换为调用 HoneyGUI API 的 C 代码
- 使用 SCons 编译用户代码和 HoneyGUI SDK
- 在 PC 上运行编译后的模拟器，显示预览效果

### 实现代码

```typescript
class ProjectCreator {
  async createProject(config: ProjectConfig) {
    const projectPath = path.join(config.saveLocation, config.projectName);
    const sdkPath = config.honeyguiSdkPath;
    
    // 1. 创建目录结构
    this.createDirectories(projectPath);
    
    // 2. 拷贝模拟器入口
    await this.copySimulatorEntry(sdkPath, projectPath);
    
    // 3. 自定义 main.c
    await this.customizeMainC(projectPath, config);
    
    // 4. 生成 SConstruct
    this.generateSConstruct(projectPath, config);
    
    // 5. 生成初始 HML
    this.generateInitialHML(projectPath, config);
    
    // 6. 生成编译脚本
    this.generateBuildScripts(projectPath);
    
    // 7. 保存项目配置
    this.saveProjectConfig(projectPath, config);
  }
  
  private async copySimulatorEntry(sdkPath: string, projectPath: string) {
    const simSrcPath = path.join(sdkPath, 'win32_sim');
    const simDestPath = path.join(projectPath, 'win32_sim');
    
    // 拷贝必要的文件和目录
    const itemsToCopy = [
      'main.c',              // 主入口
      'port/',               // 平台适配层
      'RTE/',                // SDL2 等依赖库
    ];
    
    for (const item of itemsToCopy) {
      const src = path.join(simSrcPath, item);
      const dest = path.join(simDestPath, item);
      
      if (fs.statSync(src).isDirectory()) {
        // 递归拷贝目录
        fs.cpSync(src, dest, { recursive: true });
      } else {
        // 拷贝文件
        fs.copyFileSync(src, dest);
      }
    }
  }
  
  private async customizeMainC(projectPath: string, config: ProjectConfig) {
    const mainCPath = path.join(projectPath, 'win32_sim/main.c');
    let content = fs.readFileSync(mainCPath, 'utf-8');
    
    // 替换窗口配置
    const [width, height] = config.resolution.split('X');
    content = content.replace(
      /#define WINDOW_WIDTH\s+\d+/,
      `#define WINDOW_WIDTH  ${width}`
    );
    content = content.replace(
      /#define WINDOW_HEIGHT\s+\d+/,
      `#define WINDOW_HEIGHT ${height}`
    );
    content = content.replace(
      /#define WINDOW_TITLE\s+"[^"]*"/,
      `#define WINDOW_TITLE  "${config.projectName} - HoneyGUI Preview"`
    );
    
    fs.writeFileSync(mainCPath, content);
  }
}
```

### 可定制的 main.c 模板

```c
// win32_sim/main.c
#include "gui_api.h"
#include "gui_server.h"
#include <SDL2/SDL.h>

// 项目配置（由生成器自动填充）
#define WINDOW_WIDTH  480
#define WINDOW_HEIGHT 272
#define WINDOW_TITLE  "NewProject - HoneyGUI Preview"

int main(int argc, char *argv[]) {
    // 1. 初始化 SDL2
    if (SDL_Init(SDL_INIT_VIDEO) < 0) {
        printf("SDL2 初始化失败: %s\n", SDL_GetError());
        return -1;
    }
    
    // 2. 创建窗口
    SDL_Window *window = SDL_CreateWindow(
        WINDOW_TITLE,
        SDL_WINDOWPOS_CENTERED,
        SDL_WINDOWPOS_CENTERED,
        WINDOW_WIDTH,
        WINDOW_HEIGHT,
        SDL_WINDOW_OPENGL | SDL_WINDOW_SHOWN
    );
    
    if (!window) {
        printf("窗口创建失败: %s\n", SDL_GetError());
        SDL_Quit();
        return -1;
    }
    
    // 3. 初始化 OpenGL 上下文
    SDL_GLContext gl_context = SDL_GL_CreateContext(window);
    if (!gl_context) {
        printf("OpenGL 上下文创建失败: %s\n", SDL_GetError());
        SDL_DestroyWindow(window);
        SDL_Quit();
        return -1;
    }
    
    // 4. 初始化 HoneyGUI
    gui_server_init();
    
    // 5. 用户界面会通过 GUI_VIEW_INSTANCE 自动注册
    
    // 6. 主循环
    bool running = true;
    SDL_Event event;
    
    while (running) {
        // 处理事件
        while (SDL_PollEvent(&event)) {
            if (event.type == SDL_QUIT) {
                running = false;
            }
            // 传递事件给 HoneyGUI
            gui_server_handle_event(&event);
        }
        
        // 更新和渲染
        gui_server_update();
        gui_server_render();
        
        SDL_GL_SwapWindow(window);
        SDL_Delay(16);  // ~60 FPS
    }
    
    // 7. 清理
    SDL_GL_DeleteContext(gl_context);
    SDL_DestroyWindow(window);
    SDL_Quit();
    
    return 0;
}
```

**用户可定制的部分**：
- 窗口大小和标题（自动生成）
- 初始化代码（添加自定义初始化）
- 事件处理（添加自定义事件）
- 主循环逻辑（添加调试代码）
- FPS 控制（调整帧率）

## 编译入口：SConstruct

### 自动生成的 SConstruct

```python
import os
import sys

# HoneyGUI SDK 路径（只用于链接 realgui 库）
HONEYGUI_SDK = '/home/howie_wang/workspace/HoneyGUI'

# 项目路径
PROJECT_PATH = os.getcwd()

# 添加 HoneyGUI 工具路径
sys.path.append(os.path.join(HONEYGUI_SDK, 'tool/scons-tool'))

# 导入 HoneyGUI 构建工具
from building import *

# 环境配置
env = Environment()

# 设置编译器
env['CC'] = 'gcc'
env['CXX'] = 'g++'

# 编译标志
env.Append(CFLAGS=['-std=c99', '-O2'])
env.Append(CPPPATH=[
    os.path.join(HONEYGUI_SDK, 'realgui'),
    os.path.join(HONEYGUI_SDK, 'realgui/server'),
    os.path.join(HONEYGUI_SDK, 'realgui/widget'),
    os.path.join(HONEYGUI_SDK, 'realgui/engine'),
    os.path.join(HONEYGUI_SDK, 'realgui/core'),
    os.path.join(PROJECT_PATH, 'win32_sim/port'),  # 使用项目内的 port
    'src',
])

# 链接库路径（使用项目内的 SDL2）
SDL2_LIB_PATH = os.path.join(PROJECT_PATH, 'win32_sim/RTE/SDL2-2.26.0/x86_64-w64-mingw32/lib')
env.Append(LIBPATH=[SDL2_LIB_PATH])
env.Append(LIBS=['SDL2main', 'SDL2', 'opengl32', 'mingw32'])

# 收集源文件
# 1. HoneyGUI SDK 源文件（只链接 realgui 库）
honeygui_src = Glob(os.path.join(HONEYGUI_SDK, 'realgui/**/*.c'))

# 2. 项目内的模拟器入口和适配层
sim_src = []
sim_src.append(os.path.join(PROJECT_PATH, 'win32_sim/main.c'))
sim_src += Glob(os.path.join(PROJECT_PATH, 'win32_sim/port/**/*.c'))

# 3. 用户生成的代码
user_src = Glob('src/**/*.c')

# 合并所有源文件
all_src = sim_src + honeygui_src + user_src

# 构建可执行文件
env.Program('build/gui', all_src)
```

**关键点**：
- HoneyGUI SDK 路径只用于链接 `realgui` 库源码
- 模拟器入口 `main.c` 和平台适配层 `port/` 使用项目内的文件
- SDL2 等依赖库也在项目内，不依赖 SDK 的特定位置

## Runner 定义

### Runner 是什么

**Runner = 编译后的模拟器可执行文件**

- Windows: `build/gui.exe`
- Linux: `build/gui`

这是由 HoneyGUI SDK 的 win32_sim 和用户代码编译出来的可执行程序。

### Runner 的职责

#### 1. 图形渲染
- 创建 SDL2 窗口
- 初始化 OpenGL 上下文
- 渲染 GUI 界面到窗口

#### 2. 执行用户代码
- 加载并执行生成的 C 代码（`NewProjectMain.c`）
- 调用 `GUI_VIEW_INSTANCE` 注册的视图
- 执行 `switch_in` 回调创建组件

#### 3. 事件处理
- 捕获鼠标点击、移动
- 捕获键盘输入
- 传递给 HoneyGUI 事件系统

#### 4. 资源加载
- 从 `assets/` 目录加载图片、字体
- 解码图片格式（PNG、JPG等）
- 管理资源缓存

#### 5. 运行循环
```c
// Runner 的主循环（简化版）
int main() {
    // 1. 初始化 SDL2 和 OpenGL
    SDL_Init(SDL_INIT_VIDEO);
    SDL_Window *window = SDL_CreateWindow(...);
    
    // 2. 初始化 HoneyGUI
    gui_server_init();
    
    // 3. 加载用户界面（调用生成的代码）
    // GUI_VIEW_INSTANCE 会自动注册
    
    // 4. 主循环
    while (running) {
        // 处理事件
        SDL_Event event;
        while (SDL_PollEvent(&event)) {
            handle_input(&event);
        }
        
        // 更新 GUI
        gui_server_update();
        
        // 渲染
        gui_server_render();
        SDL_GL_SwapWindow(window);
    }
    
    // 5. 清理
    SDL_Quit();
    return 0;
}
```

#### 1. 进程管理
- 启动 Runner 进程

- 处理编译错误

#### 3. 输出监听

- 显示在 VSCode 输出面板

#### 4. 热重载协调
- 检测文件变化
- 重新生成代码

- 重启 Runner

#### 5. 状态管理
- 更新状态栏
- 显示进度提示

## 优势总结

1. **真实预览**：使用真实的 HoneyGUI 引擎，预览效果与最终设备一致
2. **快速迭代**：增量编译 + 热重载，修改后1-3秒即可看到效果
3. **跨平台**：Windows 和 Linux 都支持
4. **无需硬件**：完全在 PC 上开发和测试
5. **项目完全独立**：
   - 包含所有必需的代码和依赖库
   - 可以脱离插件独立编译运行
   - 不依赖 SDK 的特定目录结构
6. **SDK 保持干净**：
   - 模拟器入口拷贝到项目内
   - 不会污染 HoneyGUI SDK
   - 多个项目互不影响
7. **可定制化**：
   - 用户可以修改 main.c 添加自定义初始化
   - 可以调整窗口大小、标题、FPS 等
   - 可以添加项目特定的调试代码
8. **易于调试**：可以使用标准调试工具调试生成的 C 代码
9. **配置清晰**：SDK 路径在项目创建时指定，保存在项目配置中
10. **版本隔离**：每个项目使用自己的模拟器入口版本，升级 SDK 不影响现有项目
11. **易于分享**：项目包含所有依赖，可以独立分发给其他开发者
